name: Node.js CI (safe / skip if no Node project)

on:
  push:
  pull_request:

jobs:
  build:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        node-version: [18.x, 20.x, 22.x]

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Quick check for Node project files
        id: check
        run: |
          if [ -f package.json ]; then
            echo "node_project=true" >> $GITHUB_OUTPUT
          else
            echo "node_project=false" >> $GITHUB_OUTPUT
          fi

      - name: Skip CI if no package.json
        if: steps.check.outputs.node_project == 'false'
        run: |
          echo "No package.json found. Skipping Node build (this is expected for non-Node repos)."
          exit 0

      - name: Use Node.js ${{ matrix.node-version }}
        if: steps.check.outputs.node_project == 'true'
        uses: actions/setup-node@v4
        with:
          node-version: ${{ matrix.node-version }}

      - name: Install dependencies (use lockfile if present)
        if: steps.check.outputs.node_project == 'true'
        run: |
          if [ -f package-lock.json ]; then
            npm ci
          elif [ -f yarn.lock ]; then
            yarn install --frozen-lockfile || yarn install
          else
            npm install
          fi

      - name: Build (if present)
        if: steps.check.outputs.node_project == 'true'
        run: npm run build --if-present

      - name: Test (if present)
        if: steps.check.outputs.node_project == 'true'
        run: npm test --if-present
